-- Crear base de datos CREATE DATABASE gategourmet; USE gategourmet; -- Crear tabla usuarios CREATE TABLE usuarios ( correo VARCHAR(100) NOT NULL UNIQUE, nombres_apellidos VARCHAR(100) NOT NULL, nombre_usuario VARCHAR(50) NOT NULL PRIMARY KEY, contrasena VARCHAR(255) NOT NULL, area VARCHAR(50) NOT NULL, cargo VARCHAR(50) NOT NULL, rol ENUM('Administrador', 'Aprobador', 'Digitador', 'Observador') NOT NULL, estado ENUM('Activo', 'Inactivo') DEFAULT 'Activo' ); -- Crear tabla administradores CREATE TABLE administradores ( correo VARCHAR(100) NOT NULL UNIQUE, nombres_apellidos VARCHAR(100) NOT NULL, nombre_usuario VARCHAR(50) NOT NULL PRIMARY KEY, contrasena VARCHAR(255) NOT NULL, area VARCHAR(50) NOT NULL, cargo VARCHAR(50) NOT NULL, rol ENUM('Administrador', 'Aprobador', 'Digitador', 'Observador') NOT NULL, estado ENUM('Activo', 'Inactivo') DEFAULT 'Activo' ); -- Crear tabla cargos CREATE TABLE cargos ( id INT(11) NOT NULL AUTO_INCREMENT, nombre_cargo VARCHAR(255) NOT NULL, PRIMARY KEY (id), UNIQUE (nombre_cargo) ); -- Crear tabla roles CREATE TABLE roles ( id INT(11) NOT NULL AUTO_INCREMENT, nombre_rol VARCHAR(255) NOT NULL, PRIMARY KEY (id), UNIQUE (nombre_rol) ); CREATE TABLE correos ( id INT(11) AUTO_INCREMENT PRIMARY KEY, asunto VARCHAR(255), remitente VARCHAR(255), fecha VARCHAR(100), cuerpo TEXT ); -- Crear tabla documentos CREATE TABLE documentos ( id INT(11) NOT NULL AUTO_INCREMENT, codigo VARCHAR(50) NOT NULL, tipo_documento ENUM('Word', 'Excel', 'PDF', 'PowerPoint', 'Texto', 'Otro') NOT NULL, estado ENUM('Activo', 'Inactivo', 'Pendiente', 'Archivado') NOT NULL, fecha_creacion TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP, fecha_modificacion TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, PRIMARY KEY (id) ); -- Crear tabla inactivos CREATE TABLE inactivos ( correo VARCHAR(100) NOT NULL UNIQUE, nombres_apellidos VARCHAR(100) NOT NULL, nombre_usuario VARCHAR(50) NOT NULL PRIMARY KEY, contrasena VARCHAR(255) NOT NULL, area VARCHAR(50) NOT NULL, cargo VARCHAR(50) NOT NULL, rol ENUM('Administrador','Aprobador', 'Digitador', 'Observador') NOT NULL, estado ENUM('Activo', 'Inactivo') DEFAULT 'Inactivo' ); -- Crear tabla listado_maestro CREATE TABLE listado_maestro ( id INT AUTO_INCREMENT PRIMARY KEY, proceso VARCHAR(255), codigo VARCHAR(255), titulo_documento VARCHAR(255), tipo VARCHAR(255), version VARCHAR(50), estado VARCHAR(50), fecha_aprobacion VARCHAR(50), -- Mantener como VARCHAR para formato 'dd/mmm/yyyy' areas TEXT, motivo_del_cambio TEXT, tiempo_de_retencion VARCHAR(255), responsable_de_retencion VARCHAR(255), lugar_de_almacenamiento_fisico TEXT, lugar_de_almacenamiento_magnetico TEXT, conservacion TEXT, disposicion_final TEXT, copias_controladas TEXT, fecha_de_vigencia VARCHAR(50), -- Mantener como VARCHAR para formato 'dd/mmm/yyyy' dias INT, senal_alerta VARCHAR(50), obsoleto VARCHAR(50), anulado VARCHAR(50), en_actualizacion VARCHAR(50) ) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci; -- Crear tabla log_eventos CREATE TABLE log_eventos ( id_evento INT AUTO_INCREMENT PRIMARY KEY, nombre_usuario VARCHAR(100) NOT NULL, fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP, detalle TEXT, FOREIGN KEY (nombre_usuario) REFERENCES usuarios(nombre_usuario) ON DELETE CASCADE ); -- Crear tabla movimientos CREATE TABLE movimientos ( id INT(11) NOT NULL AUTO_INCREMENT, nombre_usuario VARCHAR(255) NOT NULL, accion VARCHAR(255) NOT NULL, fecha TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP, PRIMARY KEY (id) ); -- Crear tabla password_resets CREATE TABLE password_resets ( id_password INT(11) NOT NULL AUTO_INCREMENT, correo VARCHAR(100) NOT NULL, token VARCHAR(255) NOT NULL, created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP, PRIMARY KEY (id_password), FOREIGN KEY (correo) REFERENCES usuarios(correo) ON DELETE CASCADE );
